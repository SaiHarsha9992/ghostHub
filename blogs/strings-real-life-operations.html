üîπ String Operations in C

In C, strings are arrays of characters terminated by \0. Most string functions are in .

    char str1[50] = "Hello";
    char str2[] = " World";

    // Concatenation
    strcat(str1, str2);
    printf("Concatenation: %s\n", str1);

    // Length
    printf("Length: %lu\n", strlen(str1));

    // Substring (manually extracting first 5 chars)
    char sub[6];
    strncpy(sub, str1, 5);
    sub[5] = '\0';
    printf("Substring: %s\n", sub);

    // Changing case (to uppercase)
    for(int i=0; str1[i]; i++) {
        str1[i] = toupper(str1[i]);
    }
    printf("Uppercase: %s\n", str1);

    // Searching
    if (strstr(str1, "WORLD")) {
        printf("Found WORLD in string\n");
    }

    return 0;
}

üîπ String Operations in C++

In C++, we have the std::string class which makes string handling easier.

#include 
#include 
#include 
using namespace std;

int main() {
    string str1 = "Hello";
    string str2 = " World";

    // Concatenation
    string result = str1 + str2;
    cout << "Concatenation: " << result << endl;

    // Length
    cout << "Length: " << result.length() << endl;

    // Substring
    cout << "Substring (0-5): " << result.substr(0, 5) << endl;

    // Changing case (to lowercase)
    transform(result.begin(), result.end(), result.begin(), ::tolower);
    cout << "Lowercase: " << result << endl;

    // Searching
    if (result.find("world") != string::npos) {
        cout << "Found 'world' in string" << endl;
    }

    return 0;
}

üîπ String Operations in Java

Java provides the String class, which is immutable, and methods in the String class.

public class StringOperations {
    public static void main(String[] args) {
        String str1 = "Hello";
        String str2 = " World";

        // Concatenation
        String result = str1 + str2;
        System.out.println("Concatenation: " + result);

        // Length
        System.out.println("Length: " + result.length());

        // Substring
        System.out.println("Substring (0-5): " + result.substring(0, 5));

        // Changing case
        System.out.println("Uppercase: " + result.toUpperCase());
        System.out.println("Lowercase: " + result.toLowerCase());

        // Searching
        if (result.contains("World")) {
            System.out.println("Found 'World' in string");
        }

        // Comparison
        System.out.println("Equal? " + str1.equals(str2));
    }
}
