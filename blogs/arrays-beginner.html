Arrays For Beginners

1. What is an Array?

An array is a collection of elements of the same type stored in contiguous memory locations and accessed using an index.

üí° Think of an array like a row of lockers:

  • Same type: All slots store similar items (e.g., books only, not books + fruits).
  • Indexed: Each locker has a unique number (0, 1, 2...).
  • Fixed size: The number of lockers is decided at creation.

2. Why Learn Arrays First?

Arrays are the foundation for:

  • Matrices
  • Strings
  • Advanced data structures like stacks, queues, heaps, hash tables

3. Common Array Operations

Operation Description
Traversal Visiting each element
Access Getting an element using index
Update Changing value at a given index
Insertion Adding a value at a specific index
Deletion Removing a value from an index
Search Finding a value in the array

4. Operations

A. Traversal

Visit each element and print it.

C: Traversal

#include 
int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int n = sizeof(arr) / sizeof(arr[0]);
    for(int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}

C++: Traversal

#include 
using namespace std;
int main() {
    int arr[] = {10, 20, 30, 40, 50};
    for(int i : arr) {
        cout << i << " ";
    }
    return 0;
}

Python: Traversal

arr = [10, 20, 30, 40, 50]
for num in arr:
    print(num, end=" ")

Java: Traversal

public class TraverseArray {
    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        for (int num : arr) {
            System.out.print(num + " ");
        }
    }
}

D. Insertion

Insert element at a specific index (shifting elements).

C: Insertion

#include 
int main() {
    int arr[6] = {10, 20, 30, 40, 50};
    int n = 5, pos = 2, val = 99;
    for(int i = n; i > pos; i--) {
        arr[i] = arr[i - 1];
    }
    arr[pos] = val;
    n++;
    for(int i = 0; i < n; i++) printf("%d ", arr[i]);
    return 0;
}

C++: Insertion

#include 
using namespace std;
int main() {
    int arr[6] = {10, 20, 30, 40, 50};
    int n = 5, pos = 2, val = 99;
    for(int i = n; i > pos; i--) arr[i] = arr[i-1];
    arr[pos] = val;
    n++;
    for(int i = 0; i < n; i++) cout << arr[i] << " ";
    return 0;
}

arr = [10, 20, 30, 40, 50]

Java: Insertion

import java.util.Arrays;
public class InsertArray {
    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50, 0};
        int n = 5, pos = 2, val = 99;
        for(int i = n; i > pos; i--) arr[i] = arr[i-1];
        arr[pos] = val;
        n++;
        System.out.println(Arrays.toString(arr));
    }
}

E. Deletion

Remove element at a specific index (shifting elements).

C: Deletion

#include 
int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int n = 5, pos = 2;
    for(int i = pos; i < n-1; i++) arr[i] = arr[i+1];
    n--;
    for(int i = 0; i < n; i++) printf("%d ", arr[i]);
    return 0;
}

C++: Deletion

#include 
using namespace std;
int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int n = 5, pos = 2;
    for(int i = pos; i < n-1; i++) arr[i] = arr[i+1];
    n--;
    for(int i = 0; i < n; i++) cout << arr[i] << " ";
    return 0;
}

Python: Deletion

arr = [10, 20, 30, 40, 50]
arr.pop(2)
print(arr)

Java: Deletion

import java.util.Arrays;
public class DeleteArray {
    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        int n = arr.length, pos = 2;
        for(int i = pos; i < n-1; i++) arr[i] = arr[i+1];
        n--;
        System.out.println(Arrays.toString(Arrays.copyOf(arr, n)));
    }
}

F. Search

Find an element‚Äôs index in an array.

C: Search

#include 
int main() {
    int arr[] = {10, 20, 30, 40, 50}, key = 30;
    int n = sizeof(arr) / sizeof(arr[0]);
    for(int i = 0; i < n; i++) {
        if(arr[i] == key) {
            printf("Found at index %d", i);
            return 0;
        }
    }
    printf("Not found");
    return 0;
}

C++: Search

#include 
using namespace std;
int main() {
    int arr[] = {10, 20, 30, 40, 50}, key = 30;
    int n = sizeof(arr) / sizeof(arr[0]);
    for(int i = 0; i < n; i++) {
        if(arr[i] == key) {
            cout << "Found at index " << i;
            return 0;
        }
    }
    cout << "Not found";
    return 0;
}

Python: Search

arr = [10, 20, 30, 40, 50]
key = 30
if key in arr:
    print("Found at index", arr.index(key))
else:
    print("Not found")

Java: Search

public class SearchArray {
    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        int key = 30;
        for(int i = 0; i < arr.length; i++) {
            if(arr[i] == key) {
                System.out.println("Found at index " + i);
                return;
            }
        }
        System.out.println("Not found");
    }
}
