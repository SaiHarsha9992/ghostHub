Queues ‚Äì Intermediate Level

1. Recap of Basics

A queue is a linear data structure that follows the FIFO principle. Basic array-based queues have a limitation: once elements are dequeued, the unused memory at the front of the array cannot be easily reused. That‚Äôs why we move to a more efficient implementation: the **Circular Queue**.

2. Circular Queue

What is a Circular Queue?

A circular queue connects the rear of the array back to the front, forming a circle. This allows for efficient memory use and avoids wasting space when elements are dequeued.

üí° Real-life analogy: A musical chairs game where players keep rotating in a circle. As one person leaves, a new person can join the empty chair.

The key logic is using the modulo operator (`%`) to handle the wraparound effect of the pointers.

C Implementation

#include 
#define SIZE 5

int cq[SIZE], front = -1, rear = -1;

void enqueue(int val) {
    if ((rear + 1) % SIZE == front) {
        printf("Queue Overflow\n");
    } else {
        if (front == -1) front = 0;
        rear = (rear + 1) % SIZE;
        cq[rear] = val;
    }
}

void dequeue() {
    if (front == -1) {
        printf("Queue Underflow\n");
    } else {
        printf("Dequeued: %d\n", cq[front]);
        if (front == rear) front = rear = -1;
        else front = (front + 1) % SIZE;
    }
}

void display() {
    if (front == -1) {
        printf("Queue Empty\n");
        return;
    }
    int i = front;
    while (1) {
        printf("%d ", cq[i]);
        if (i == rear) break;
        i = (i + 1) % SIZE;
    }
    printf("\n");
}

int main() {
    enqueue(10);
    enqueue(20);
    enqueue(30);
    dequeue();
    enqueue(40);
    enqueue(50);
    display();
    return 0;
}

class CircularQueue:
    def __init__(self, size):
        self.size = size
        self.queue = [None] * size
        self.front = self.rear = -1

    def enqueue(self, val):
        if (self.rear + 1) % self.size == self.front:
            print("Queue Overflow")
        else:
            if self.front == -1:
                self.front = 0
            self.rear = (self.rear + 1) % self.size
            self.queue[self.rear] = val

    def dequeue(self):
        if self.front == -1:
            print("Queue Underflow")
        else:
            print("Dequeued:", self.queue[self.front])
            if self.front == self.rear:
                self.front = self.rear = -1
            else:
                self.front = (self.front + 1) % self.size

    def display(self):
        if self.front == -1:
            print("Queue Empty")
        else:
            i = self.front
            while True:
                print(self.queue[i], end=" ")
                if i == self.rear:
                    break
                i = (i + 1) % self.size
            print()

3. Deque (Double-Ended Queue)

What is a Deque?

A deque (pronounced "deck") is a double-ended queue that allows insertion and deletion from both ends.

üí° Real-life analogy: A double-ended line at a supermarket where customers can join or leave from either side. This gives it more flexibility than a standard queue.

Deque Example in C++ (using STL)

#include 
#include 
using namespace std;

int main() {
    deque dq;
    dq.push_back(10);   // Insert at rear
    dq.push_front(20);  // Insert at front
    dq.push_back(30);
    
    cout << "Deque: ";
    for (int x : dq) cout << x << " ";
    cout << endl;

    dq.pop_front();
    dq.pop_back();

    cout << "After popping: ";
    for (int x : dq) cout << x << " ";
    cout << endl;
    return 0;
}

Deque Example in Java (using `java.util.Deque`)

import java.util.Deque;
import java.util.LinkedList;

public class DequeExample {
    public static void main(String[] args) {
        Deque dq = new LinkedList<>();
        dq.addLast(10); // Insert at rear
        dq.addFirst(20); // Insert at front
        dq.addLast(30);
        
        System.out.print("Deque: ");
        for (int x : dq) System.out.print(x + " ");
        System.out.println();

        dq.removeFirst(); // Pop from front
        dq.removeLast(); // Pop from rear

        System.out.print("After popping: ");
        for (int x : dq) System.out.print(x + " ");
        System.out.println();
    }
}

Deque Example in C (using a Linked List)

#include 
#include 

struct Node {
    int data;
    struct Node *prev, *next;
};

struct Deque {
    struct Node *front, *rear;
};

void pushFront(struct Deque* dq, int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->prev = NULL;
    newNode->next = dq->front;
    if (dq->front != NULL) dq->front->prev = newNode;
    else dq->rear = newNode;
    dq->front = newNode;
}

void popBack(struct Deque* dq) {
    if (dq->rear == NULL) {
        printf("Deque is empty\n");
        return;
    }
    struct Node* temp = dq->rear;
    printf("Popped from rear: %d\n", temp->data);
    dq->rear = temp->prev;
    if (dq->rear != NULL) dq->rear->next = NULL;
    else dq->front = NULL;
    free(temp);
}

int main() {
    struct Deque dq = {NULL, NULL};
    pushFront(&dq, 10);
    pushFront(&dq, 20);
    popBack(&dq);
    popBack(&dq);
    return 0;
}

4. Applications of Queues (Intermediate)

  • **CPU Scheduling (Round Robin):** A circular queue is used to give each process a fair share of CPU time.
  • **Breadth-First Search (BFS):** An algorithm for traversing a graph or tree level by level.
  • **Message Buffering:** Network packets are often stored in a queue to be processed in the order they were received.
  • **Deque for Palindrome Checking:** You can use a deque to check if a string is a palindrome by adding characters from the front and rear and comparing them as you remove them.

5. Problem-Solving Strategies

  • Use a **circular queue** to optimize space in a fixed-size queue, especially when you have frequent `enqueue` and `dequeue` operations.
  • Use a **deque** for problems that require adding or removing elements from both the front and the rear, such as finding a palindrome or in a sliding window maximum problem.
